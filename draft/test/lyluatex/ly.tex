\begin{lilypond}
    MusicS = { 
        \clef "treble"
        \time 3/8
        a'8 b'8 c''8
        d''4.
        cis''4 r8
    }

    MusicB = {
        \clef "bass"
        \time 3/8
        d'4 c'8
        bes4.
        a4 r8
    }

    LyricsS = \lyricmode {
        This is a ca -- dence!
    }
    \score {
        <<
            \new Staff 
            \with { instrumentName = "Soprano" }
                << 
                \new Voice = "S" { \MusicS }
                \new Lyrics \lyricsto "S" { \LyricsS }
                >>
            \new Staff
            \with { instrumentName = "Bc." }
                <<
                \new Voice { \MusicB }
                >>
        >>
    }
\end{lilypond}
    \end{center}
\end{table}
