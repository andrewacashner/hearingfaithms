% Hidalgo, Venid querubines
% D-Mbs: Mus. ms. 2895
% Copla 5
\begin{poemtranslation}
    \begin{original}
        \Stanza{4}
        5. Las aves que sobre el viento &
        tejen plumado pensíl, &
        para hacer la salva al sol, &
        cada pico es un clarín. \&

        \Stanza{3}
        \emph{Escuchad, atended,} &
        \emph{que el clarín de su aurora,} &
        \emph{mi voz ha de ser.} \&
    \end{original}
    \begin{translation}
        \Stanza{4}
        The birds who upon the wind &
        trace their plumed quill, &
        to sound the salvo to the sun, &
        each beak is a clarion. \&

        \Stanza{3}
        \emph{Listen, hark,} &
        \emph{for the clarion of his dawn} &
        \emph{my voice shall be.} \&
    \end{translation}
\end{poemtranslation}
